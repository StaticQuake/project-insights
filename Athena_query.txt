CREATE DATABASE project_insights;
q2
CREATE EXTERNAL TABLE IF NOT EXISTS project_insights.daily_metrics (
    id INT,
    snapshot_date STRING,
    popularity DOUBLE,
    vote_average DOUBLE,
    vote_count INT
)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ','
LINES TERMINATED BY '\n'
STORED AS TEXTFILE
LOCATION 's3://project-insights-data-sem6/raw/daily_metrics/'
TBLPROPERTIES ('skip.header.line.count'='1');

q3
ALTER TABLE project_insights.daily_metrics
ADD PARTITION (snapshot_date='2026-02-20')
LOCATION 's3://project-insights-data-sem6/raw/daily_metrics/snapshot_date=2026-02-20/';

q4
SELECT * FROM project_insights.daily_metrics LIMIT 10;

SELECT 
    COUNT(*) as total_movies,
    AVG(popularity) as avg_popularity,
    MAX(popularity) as max_popularity,
    MIN(popularity) as min_popularity
FROM project_insights.daily_metrics;
